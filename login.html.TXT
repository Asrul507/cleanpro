  <script>
    // Pre-fill from remember
    window.addEventListener('DOMContentLoaded', () => {
      const saved = localStorage.getItem('cp_remember');
      if (saved) {
        const d = JSON.parse(saved);
        document.getElementById('username').value = d.username || '';
        document.getElementById('remember').checked = true;
      }

      // Check already logged in
      const session = sessionStorage.getItem('cp_session') || localStorage.getItem('cp_session');
      if (session) {
        const user = JSON.parse(session);
        redirectByRole(user.role);
      }
    });

    function togglePassword() {
      const input = document.getElementById('password');
      const btn = document.getElementById('eyeBtn');
      if (input.type === 'password') {
        input.type = 'text';
        btn.textContent = '🙈';
      } else {
        input.type = 'password';
        btn.textContent = '👁️';
      }
    }

    async function handleLogin() {
      const username = document.getElementById('username').value.trim();
      const password = document.getElementById('password').value;
      const shift = document.querySelector('input[name="shift"]:checked').value;
      const remember = document.getElementById('remember').checked;
      const errMsg = document.getElementById('errorMsg');
      const errText = document.getElementById('errorText');
      const loginBtn = document.getElementById('loginBtn');
      const loginText = document.getElementById('loginText');
      const loginSpinner = document.getElementById('loginSpinner');

      if (!username || !password) {
        errText.textContent = 'Username dan password wajib diisi.';
        errMsg.classList.add('show');
        return;
      }

      // Loading state
      loginBtn.disabled = true;
      loginText.style.display = 'none';
      loginSpinner.style.display = 'inline';
      errMsg.classList.remove('show');

      try {
        const result = await AuthService.login(username, password, shift);

        if (result.success) {
          const session = { ...result.user, shift, loginTime: new Date().toISOString() };
          const store = remember ? localStorage : sessionStorage;
          store.setItem('cp_session', JSON.stringify(session));

          if (remember) {
            localStorage.setItem('cp_remember', JSON.stringify({ username }));
          } else {
            localStorage.removeItem('cp_remember');
          }

          redirectByRole(result.user.role);
        } else {
          errText.textContent = result.message || 'Login gagal. Coba lagi.';
          errMsg.classList.add('show');
        }
      } catch (e) {
        errText.textContent = 'Koneksi bermasalah. Periksa internet.';
        errMsg.classList.add('show');
      } finally {
        loginBtn.disabled = false;
        loginText.style.display = 'inline';
        loginSpinner.style.display = 'none';
      }
    }

    function redirectByRole(role) {
      if (role === 'supervisor') {
        window.location.href = 'pages/supervisor/dashboard.html';
      } else {
        window.location.href = 'pages/leader/dashboard.html';
      }
    }

    // Enter key support
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') handleLogin();
    });
  </script>
